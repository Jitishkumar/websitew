// Configure split APKs
splits {
    abi {
        enable true
        reset()
        include "armeabi-v7a", "arm64-v8a", "x86", "x86_64"
        universalApk false
    }
}

- Use the architecture-specific APKs generated by ./gradlew assembleRelease for direct testing on devices
- For most modern phones, use the app-arm64-v8a-release.apk file
- For older devices, use the app-armeabi-v7a-release.apk file
- For emulators, use the appropriate x86 or x86_64 APK
